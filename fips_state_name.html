function checkCustomFormInfo(formID){

var fieldName = $g(this).attr("name");
var fieldReqd = $g(this).attr("required");
var fieldVal = $g(this).val().trim();
var fieldType = $g(this).attr("fieldtype");
if(fieldType){
if(fieldType == 'inputEmail'){
if(fieldVal.length > 1){
var filter = /^([\w-\.]+)@((\[[0-9]{1,3}\.[0-9]{1,3}\.[0-9]{1,3}\.)|(([\w-]+\.)+))([a-zA-Z]{2,4}|[0-9]{1,3})(\]?)$/;

$g('#'+fieldID+"_img").attr('src','/NRCS_WIDTheme/themes/html/NRCS_WIDTheme/images/correct_sign.jpg');
}else{
isFormValidArr.push(false);
$g('#'+fieldID+"_img").attr('src','/NRCS_WIDTheme/themes/html/NRCS_WIDTheme/images/blank.gif');
}
}else if(fieldReqd && fieldVal.length < 1){
isFormValidArr.push(false);
$g('#'+fieldID+"_img").attr('src','/NRCS_WIDTheme/themes/html/NRCS_WIDTheme/images/blank.gif');
}else if(!fieldReqd && fieldVal.length == 0){
$g('#'+fieldID+"_img").attr('src','/NRCS_WIDTheme/themes/html/NRCS_WIDTheme/images/blank.gif');
}
}else if(fieldType == 'inputText' && fieldReqd){
if(fieldVal.length >= 2){
isFormValidArr.push(true);
$g('#'+fieldID+"_img").attr('src','/NRCS_WIDTheme/themes/html/NRCS_WIDTheme/images/correct_sign.jpg');
}else{
isFormValidArr.push(false);
$g('#'+fieldID+"_img").attr('src','/NRCS_WIDTheme/themes/html/NRCS_WIDTheme/images/blank.gif');
}
}else if(fieldType == 'inputPhone'){
if(fieldVal.length > 1){
var intRegex = /[0-9 -()+]+$/;
if (intRegex.test(fieldVal)) {
isFormValidArr.push(true);
$g('#'+fieldID+"_img").attr('src','/NRCS_WIDTheme/themes/html/NRCS_WIDTheme/images/correct_sign.jpg');
}else{
isFormValidArr.push(false);
$g('#'+fieldID+"_img").attr('src','/NRCS_WIDTheme/themes/html/NRCS_WIDTheme/images/blank.gif');
}
}else if(fieldReqd && fieldVal.length < 1){
isFormValidArr.push(false);
$g('#'+fieldID+"_img").attr('src','/NRCS_WIDTheme/themes/html/NRCS_WIDTheme/images/blank.gif');
}else if(!fieldReqd && fieldVal.length == 0){
$g('#'+fieldID+"_img").attr('src','/NRCS_WIDTheme/themes/html/NRCS_WIDTheme/images/blank.gif');
}
}else if(fieldType == 'dropDown' && fieldReqd){
if(fieldVal.length > 0){
isFormValidArr.push(true);
$g('#'+fieldID+"_img").attr('src','/NRCS_WIDTheme/themes/html/NRCS_WIDTheme/images/correct_sign.jpg');
}else{
isFormValidArr.push(false);
$g('#'+fieldID+"_img").attr('src','/NRCS_WIDTheme/themes/html/NRCS_WIDTheme/images/blank.gif');
}
}
}
});
var isFormValid = false;
for(var i=0;i<isFormValidArr.length;i++){
if(isFormValidArr[i]){
isFormValid = true;

isFormValid = false;

}

if(isFormValid){
$g("#CF_Submit_"+formID).removeAttr('disabled');

}
